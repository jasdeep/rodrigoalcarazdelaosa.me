	d3.csv('h.csv')

	function makeChart(hdata) {
		var t = hdata.map(function(d) {return d.t});
		var h = hdata.map(function(d) {return d.h});		
		var chart = new Chart('h-t', {
		  type: 'line',
		  data: {
		    labels: t,
		    datasets: [
		      {
		        data: h,
 			    showLine: false
		      }
		    ]
		  },
		  options: {
			  scales: {
				  xAxes: [{
					  scaleLabel: {
						  display: true,
						  labelString: 't/s'
					  }
				  }],
				  yAxes: [{
					  scaleLabel: {
						  display: true,
						  labelString: 'h/km'
					  }
				  }]
			  	
			  },
		      legend: {
		      	display: false		              
		      }
		   }
		});
	}
